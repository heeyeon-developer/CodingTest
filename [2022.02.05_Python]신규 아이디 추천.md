><h1>프로그래머스_신규 아이디 추천</h1>
1. 문제의 7가지 조건을 아래의 주석대로 실행

```python
def solution(new_id):
    special = '~!@#$%^&*()=+[{]}:?,<>/'
    #대문자를 소문자로 치환
    new_id = new_id.lower()
    #-,_,. 제외한 문자 제거
    for i in range(len(special)) :
        new_id = new_id.replace(special[i],'')
    #...인것 제거
    temp = new_id[0]
    for i in range(1,len(new_id)) :
        if new_id[i] == '.' and temp[len(temp)-1] == '.' :
            continue
        else :
            temp += new_id[i]
    #.처음 마지막 제거
    if len(temp) == 0 : 
        temp = temp
    elif temp[0] == '.' :
        temp = temp[1:]
    if len(temp) == 0 :
        temp = temp    
    elif temp[len(temp)-1] == '.' :
        temp = temp[:len(temp)-1]
    #빈문자열이라면 'a' 더하기
    if len(temp) == 0 :
        temp = 'a'
    #15개까지만 자르기 마지막이 .이라면 제거
    new_id = temp[:15]
    if len(new_id) == 0 :
        new_id = new_id    
    elif new_id[len(new_id)-1] == '.' :
        new_id = new_id[:len(new_id)-1]
    #만약 길이가 2이하라면 마지막 문자를 길이 3 될때까지 만들기
    if len(new_id) <= 2 :
        temp = new_id[len(new_id)-1]
        for i in range(3) :
            new_id += temp
            if len(new_id) == 3 :
                break
    return new_id
```
