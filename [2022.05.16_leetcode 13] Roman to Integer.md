좀 더 간단한 알고리즘으로 바꿔보기
```java
class Solution {
    public int romanToInt(String s) {
        int answer = 0;
        int hap = 0;
        int i = 0;
        char[] myS = s.toCharArray();
        boolean check = true;
        HashMap<Character, Integer> str = new HashMap<Character, Integer>();
        str.put('I',1);
        str.put('V',5); 
        str.put('X',10);
        str.put('L',50); 
        str.put('C',100);
        str.put('D',500);
        str.put('M',1000);  
        
        for(i=0; i<s.length()-1; i++){
            hap = 0;
            check = true;
            if(myS[i] == 'I' && (myS[i+1] == 'V' || myS[i+1] == 'X')){
                hap = str.get(myS[i+1])-1;
                i++;
                check = false;
            }else if(myS[i] == 'X' && (myS[i+1] == 'L' || myS[i+1] == 'C')){
                hap = str.get(myS[i+1])-10;
                i++;
                check = false;
            }else if(myS[i] == 'C' && (myS[i+1] == 'D' || myS[i+1] == 'M')){
                hap = str.get(myS[i+1])-100;
                i++;
                check = false;
            }
            if(hap == 0)
                hap = str.get(myS[i]);
            
             answer += hap;   
        }
        
        if(check)
            answer += str.get(myS[i]);
        
        return answer;
    }
}
```
